function get_main_title() {
　if ( is_singular( 'post' ) ): //個別の投稿の場合に表示
　　$category_obj = get_the_category();
　　return $category_obj[0]->name; //投稿のカテゴリー名を出力
　elseif ( is_page() ): //固定ページの場合に表示
　　return get_the_title();
　elseif ( is_category() ): //カテゴリーページの場合に表示
　　return single_cat_title(); //現在のカテゴリー名を出力
　endif;

　return '';
}

// 子ページを取得する関数
function get_child_pages( $number = -1 ) {
　$parent_id = get_the_ID();
　$args = array(
　　'posts_per_page' => $number, //取得したい記事数を指定
　　'post_type' => 'page', //取得したい投稿タイプを指定

　);

　return $child_pages;
}

function get_main_image() {
　if ( is_page() ): //固定ページの場合、サムネイルを表示

　elseif ( is_category( 'news' ) || is_singular( 'post' ) ): //「ニュース一覧」または「ニュース詳細」で表示

　　return '<img src="'. get_template_directory_uri(). '/assets/images/bg-page-dummy.png" />';
　endif;
}

// 子ページを取得する関数（※トップページ用に関数を修正）
function get_child_pages( $number = -1, $specified_id = null ) {
　if ( isset( $specified_id ) ): //「$specified_id」が指定されている場合に表示
　　$parent_id = $specified_id;
　else:
　　$parent_id = get_the_ID();
　endif;
　$args = array(
　　'posts_per_page' => $number,
　　'post_type' => 'page',
　　'orderby' => 'menu_order',
　　'order' => 'ASC',
　　'post_parent' => $parent_id,
　);
　$child_pages = new WP_Query( $args );
　return $child_pages;
}

// 特定の記事を抽出する関数
function get_specific_posts( $post_type, $taxonomy = null, $term = null, $number = -1 ) {
　$args = array(
　　'post_type' => $post_type,
　　'tax_query' => array(
　　　array(
　　　　'taxonomy' => $taxonomy,
　　　　'field' => 'slug',
　　　　'terms' => $term,
　　　),
　　),
　　'posts_per_page' => $number,
　);
　$specific_posts = new WP_Query( $args );
　return $specific_posts;
}
